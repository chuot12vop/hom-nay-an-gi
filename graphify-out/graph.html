Luồng dữ liệu & màn hình

Từ Google Sheet → lọc → danh sách món sang vòng quay.

flowchart TB
  subgraph boot["Khởi động"]
    M["main · HomNayAnGiApp"] --> BH["BaseHomePage"]
    BH --> CF["ChooseFoodScreen"]
  end
  subgraph data["Dữ liệu"]
    CF --> LD["_loadData"]
    LD --> GSS["GoogleSheetService"]
    GSS -->|"lỗi"| FB["Fallback"]
    GSS --> OK["SheetData / models"]
    FB --> OK
    OK --> FS["FilterStorageService · food_filters.json"]
  end
  subgraph user["Lọc"]
    FS --> UI["meal · giá · dị ứng"]
    UI --> BTN["Xác nhận · SavedFilters"]
    BTN --> FF["_filterFoods"]
  end
  subgraph out["Kết quả"]
    FF --> CB["onFoodsFiltered"]
    CB --> WH["onOpenWheelRequested → tab Vòng quay"]
    WH --> SW["SpinWheelScreen"]
  end
